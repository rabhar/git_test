regexbased: {
    xpathHighlight: null,
    loadui: function() {
      var that = this;
      var notes =
        "<note>Please ensure that all regex based titles are highlighted</note>";

      if (storageCache.website.pe.regexbasedXpath.length) {
        Ui.loadUi(
          "minion_select.png",
          "Are all the regexbased links highlighted ? ",
          "yes",
          "no",
          "Toggle highlighted urls",
          notes,
          function() {
            that.xpathHighlighter = new XpathHighlighter(
              storageCache.website.pe.regexbasedXpath
            );
            $("#common_container_ext").hide();
            that.secondButton();
          }
        );
      }
    }
}
